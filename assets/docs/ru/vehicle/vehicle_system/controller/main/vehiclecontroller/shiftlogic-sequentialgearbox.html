Описание

Контроллер логики переключения последовательной коробки передач управляет всеми помощниками вождения и связанными системами при использовании последовательной КПП, включая автоматическое сцепление, отсечку газа и переключение передач в аркадном режиме.

Он является расширением контроллера транспортного средства и автоматически вызывается, если автомобиль оснащён последовательной трансмиссией .

"vehicleController": {
    "calculateOptimalLoadShiftPoints": true,
    "shiftDownRPMOffsetCoef":1.20,
    "aggressionHoldOffThrottleDelay":3,
    "lowShiftDownRPM":[0,0,0,2000,2200,2200,2200,2200],
    "lowShiftUpRPM":[0,0,3900,3800,3700,3700,3700],
    "clutchLaunchStartRPM": 3000,
    "clutchLaunchTargetRPM": 2000,
    "ignitionCutTime": 0.09
},
